💻 USAGE EXAMPLE:

import { useMagicLink } from '@/hooks/use-magic-link'

function MyApp() {
  const { user, isAuthenticated, signOut } = useMagicLink()

  if (!isAuthenticated) {
    return <a href="/auth/magic-link">Sign In</a>
  }

  return (
    <div>
      <p>Welcome, {user?.email}!</p>
      <button onClick={signOut}>Sign Out</button>
    </div>
  )
}
